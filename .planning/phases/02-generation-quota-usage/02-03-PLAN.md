```python
import logging

from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.database import get_db
from app.models import User, UsageLog
from app.schemas import GenerateRequest, GenerateResponse, UsageDetail
from app.providers.base import BaseProvider
from app.providers.claude import get_provider

logger = logging.getLogger(__name__)

router = APIRouter(prefix="/users", tags=["generate"])


@router.post("/{user_id}/generate", response_model=GenerateResponse, status_code=200)
async def generate(
    user_id: int,
    body: GenerateRequest,
    db: AsyncSession = Depends(get_db),
    provider: BaseProvider = Depends(get_provider),
):
    # --- estimate (sync, outside any TX) ---
    estimated_tokens = provider.estimate_tokens(body.prompt)

    # --- RESERVE TX: SELECT FOR UPDATE, quota check, increment reserved_credits ---
    async with db.begin():
        result = await db.execute(
            select(User).where(User.id == user_id).with_for_update()
        )
        user = result.scalar_one_or_none()
        if user is None:
            raise HTTPException(status_code=404, detail=f"User {user_id} not found")

        # 402 if user somehow has quota=0 and was never configured (treat as no config)
        # quota column is NOT NULL (set on create), so null-config = user doesn't exist
        estimated_credits = int(estimated_tokens * user.multiplier)

        remaining = user.quota - user.used_credits - user.reserved_credits
        if estimated_credits > remaining:
            # Write quota_exceeded log before returning error (D-22)
            db.add(UsageLog(
                user_id=user_id,
                prompt_tokens=estimated_tokens,
                completion_tokens=0,
                estimated_credits=None,
                actual_credits=None,
                status="quota_exceeded",
            ))
            # commit happens at end of async with db.begin() block
            raise HTTPException(status_code=402, detail="Quota exceeded")

        user.reserved_credits += estimated_credits
        # TX commits here (end of async with block), lock released

    reserved = True
    settled = False
    generation_result = None

    try:
        # --- AI GENERATION (outside any TX) ---
        generation_result = await provider.generate(body.prompt)

        # --- SETTLE TX: SELECT FOR UPDATE, debit actual, release reservation ---
        actual_credits = int(
            (generation_result.prompt_tokens + generation_result.completion_tokens)
            * user.multiplier
        )
        async with db.begin():
            result = await db.execute(
                select(User).where(User.id == user_id).with_for_update()
            )
            user = result.scalar_one_or_none()
            # user must exist here (we just checked above); skip None guard
            user.reserved_credits -= estimated_credits
            user.used_credits += actual_credits

            db.add(UsageLog(
                user_id=user_id,
                prompt_tokens=generation_result.prompt_tokens,
                completion_tokens=generation_result.completion_tokens,
                estimated_credits=estimated_credits,
                actual_credits=actual_credits,
                status="success",
            ))
            # TX commits here

        settled = True

    except HTTPException:
        raise  # quota_exceeded already raised inside reserve block (unreachable here, but safe)

    except Exception as exc:
        # AI error OR settle TX error — log and return 503
        logger.error(
            "generate failed for user_id=%s: %s",
            user_id,
            exc,
            exc_info=True,
        )
        # Write ai_error log (D-22: before returning error response)
        try:
            async with db.begin():
                db.add(UsageLog(
                    user_id=user_id,
                    prompt_tokens=estimated_tokens,
                    completion_tokens=0,
                    estimated_credits=None,
                    actual_credits=None,
                    status="ai_error",
                ))
        except Exception:
            logger.error("Failed to write ai_error UsageLog for user_id=%s", user_id)

        raise HTTPException(status_code=503, detail="AI generation failed")

    finally:
        if reserved and not settled:
            # Release reservation — undo the reserved_credits increment (D-16)
            try:
                async with db.begin():
                    result = await db.execute(
                        select(User).where(User.id == user_id).with_for_update()
                    )
                    u = result.scalar_one_or_none()
                    if u is not None:
                        u.reserved_credits -= estimated_credits
            except Exception:
                logger.error(
                    "Failed to release reservation for user_id=%s estimated_credits=%s",
                    user_id,
                    estimated_credits,
                )

    return GenerateResponse(
        text=generation_result.text,
        usage=UsageDetail(
            prompt_tokens=generation_result.prompt_tokens,
            completion_tokens=generation_result.completion_tokens,
            estimated_credits=estimated_credits,
            actual_credits=actual_credits,
        ),
    )
```

Implementation notes:
- `async with db.begin()` creates a nested transaction context; SQLAlchemy async sessions
  support this — the block commits on exit and rolls back on exception.
- user.multiplier is read from the first SELECT FOR UPDATE; the settle TX re-fetches user
  to prevent stale reads on reserved_credits/used_credits, but uses the original multiplier
  for actual_credits (multiplier doesn't change mid-request, USAGE-03 compliance).
- The finally block releases reservation when `reserved=True and settled=False`. This covers:
  AI error (generation_result is None), settle TX failure (exception before settled=True),
  any unexpected exception. It does NOT run if quota_exceeded (reserved stays False).
- ai_error log write is best-effort (wrapped in try/except) — if DB is down, we can't log
  but we still return 503.
- No prompt text stored in UsageLog per D-20.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP client → POST /users/{id}/generate | Untrusted: user_id path param + prompt body |
| app → Postgres (reserve TX) | SELECT FOR UPDATE on User row |
| app → AI provider | prompt string leaves process boundary |
| app → Postgres (settle TX) | Writes actual token counts and UsageLog row |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-03-01 | Spoofing | user_id path parameter | mitigate | DB lookup confirms user exists (404 if not); no auth in this service per scope — reviewer aware |
| T-02-03-02 | Tampering | prompt body (unbounded string) | mitigate | min_length=1 validated by Pydantic before endpoint body; no max length — bounded by provider estimate before quota check (cost is checked, not prompt size) |
| T-02-03-03 | Repudiation | UsageLog records every request | mitigate | Log written for success, quota_exceeded, ai_error — full audit trail; no-prompt policy (D-20) means no PII stored |
| T-02-03-04 | Information Disclosure | 503 error detail includes exc string | mitigate | logger.error logs exc_info=True for ops; HTTPException detail is "AI generation failed" (generic) — no internal state leaked to client |
| T-02-03-05 | Denial of Service | concurrent requests serialize at DB lock (SELECT FOR UPDATE) | accept | By design: QUOTA-03 requires serialization; lock held only during reserve check (~ms), not during AI generation |
| T-02-03-06 | Elevation of Privilege | estimated_credits overflow (int * float multiplier) | mitigate | `int(...)` cast truncates; BigInteger column holds result; Python int is arbitrary precision — no overflow possible |
| T-02-03-07 | Denial of Service | AI generation hangs indefinitely | accept | Demo scope; no timeout imposed on provider.generate(); add httpx timeout if throughput matters |
| T-02-03-08 | Tampering | settle TX failure leaves actual tokens unaccounted | accept | D-17 explicitly documents as known limitation (undercharge); logged at ERROR for ops visibility; no retry per ponytail |
</threat_model>
